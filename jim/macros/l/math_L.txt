# Macros para el modelo L basadas en los ejemplos del libro ...

defmacro suma # x1, x2
		Y <- X1
		Z <- X2
	[B]	IF Z != 0 GOTO A
		GOTO E
	[A]	Z <- Z - 1
		Y <- Y + 1
		GOTO B
endmacro

defmacro producto # x1, x2
		Z2 <- X2
	[B]	IF Z2 != 0 GOTO A
		GOTO E
	[A]	Z2 <- Z2 - 1
		Z1 <- X1 + Y
		Y <- Z1
		GOTO B
endmacro

defmacro restaParcial # x1, x2
		Y <- X1
		Z <- X2
	[C]	IF Z != 0 GOTO A
		GOTO E
	[A]	IF Y != 0 GOTO B
		GOTO A
	[B]	Y <- Y - 1
		Z <- Z - 1
		GOTO C
endmacro
