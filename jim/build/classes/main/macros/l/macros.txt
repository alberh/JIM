# pruebas de recursividad
defmacro macroA
	z1 <- macroB(x1)
	z2 <- macroC(x1)
	z3 <- macroD(x1)
	y <- z1 + z2
	y <- y + z3
endmacro

defmacro macroB
	z1 <- x1 + 1
	y <- macroD(z1)
endmacro

defmacro macroC
	y <- macroD(x1)
endmacro

defmacro macroD
	y <- x1 + 1
endmacro
